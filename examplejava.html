1 - Reverse a String function reverseString(str) { return str.split('').reverse().join(''); } reverseString("hello"); 2 - Factorialize a Number function factorialize(num) { if (num < 0) return -1; else if (num == 0) return 1; else { return (num * factorialize(num - 1)); } } factorialize(5); 3 - Check for Palindromes function palindrome(str) { var clean = str.replace(/[\W_]/g, '').toLowerCase(); var strReverse = clean.split('').reverse().join(''); if(strReverse === clean){ return true; } else { return false; } } palindrome("eye"); 4 - Title Case a Sentence function titleCase(str) { var convertToArray = str.toLowerCase().split(" "); var result = convertToArray.map(function(val){ return val.replace(val.charAt(0), val.charAt(0).toUpperCase()); }); return result.join(" "); } titleCase("I'm a little tea pot"); 5 - Return Largest Numbers in Arrays function largestOfFour(arr) { var array = arr.map(function(large){ return Math.max.apply(null, large); }); return array; } largestOfFour([[4, 5, 1, 3], [13, 27, 18, 26], [32, 35, 37, 39], [1000, 1001, 857, 1]]); 6 - Confirm the Ending function confirmEnding(str, target) { return str.substr(str.length - target.length) === target; } confirmEnding("Bastian", "n");
